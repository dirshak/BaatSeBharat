@st.cache_data
def load_topic_dataset():

    df = pd.read_csv(
        TOPIC_DATASET
    )

    df["date"] = pd.to_datetime(
        df["date"],
        errors="coerce"
    )

    df["sentiment"] = (
        df["positive"]
        - df["negative"]
    )

    df["title"] = (
        df["filename"]
        .str.replace(".txt", "")
    )

    return df.dropna(
        subset=["date"]
    )


@st.cache_data
def load_topics():

    return pd.read_csv(
        FINAL_TOPICS
    )


@st.cache_data
def load_nmf_topics():

    return pd.read_csv(
        NMF_TOPICS
    )


@st.cache_data
def load_impact_data():

    if os.path.exists(
        IMPACT_CSV
    ):
        return pd.read_csv(
            IMPACT_CSV
        )

    return pd.DataFrame()


@st.cache_data
def load_report():

    if os.path.exists(
        REPORT_PATH
    ):
        with open(
            REPORT_PATH,
            "r",
            encoding="utf-8"
        ) as f:
            return f.read()

    return ""


# ==========================================================
# HELPERS
# ==========================================================

def parse_report_sections(text):

    sections = []

    raw_sections = text.split(
        "------------------------------------------------"
    )

    for sec in raw_sections:
        if "EVENT" in sec:
            sections.append(sec.strip())

    return sections
